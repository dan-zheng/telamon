pub unsafe extern "C" fn telamon_ir_function_add_instruction(
    function: *mut Function, 
    operator: *mut Operator, 
    dimensions: *const DimId, 
    num_dimensions: usize, 
    inst_id: *mut InstId
) -> TelamonStatus

Adds an instruction performing the given operator in the given dimensions to the function. Writes the unique identifier of the instruction in inst_id. Returns TelamonStatusOk except if an error occurs. Takes ownership of the operator but does not keeps any reference to dimensions.
